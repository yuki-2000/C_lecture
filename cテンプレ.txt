int max2(int a, int b)
{
	return (a > b) ? a : b;
}

int max4(int a, int b, int c, int d)
{
	return max2(max2(a,b), max2(c,d));
}

int  sumup(int n)
{
	int sum = 0;
	int i;
	for(i = 1; i <= n; i++)
		sum += i;
	return sum;
}
